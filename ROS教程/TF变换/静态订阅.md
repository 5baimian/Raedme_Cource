/*  
    订阅坐标系信息，生成一个相对于 子级坐标系的坐标点数据，转换成父级坐标系中的坐标点,实现TF坐标转换

    实现流程:
        1.包含头文件
        2.初始化 ROS 节点
        3.创建 TF 订阅节点
        4.生成一个坐标点(相对于子级坐标系)
        5.转换坐标点(相对于父级坐标系)，转换算法被封装在TF内置函数中
        6.spin()
*/
//1.包含头文件
#include "ros/ros.h"
#include "tf2_ros/transform_listener.h"//创建订阅对象
#include "tf2_ros/buffer.h"//缓存对象
#include "geometry_msgs/PointStamped.h"//坐标点消息
#include "tf2_geometry_msgs/tf2_geometry_msgs.h"//坐标转换函数

int main(int argc, char *argv[])
{
    setlocale(LC_ALL,"");//编码
    // 2.初始化 ROS 节点
    ros::init(argc,argv,"tf_sub");
    ros::NodeHandle nh;
    // 3.创建 TF 订阅节点
    tf2_ros::Buffer buffer;//监听对象可以将订阅的数据存入buffer中
    tf2_ros::TransformListener listener(buffer);
    
    // 4.生成一个坐标点(相对于子级坐标系)
    geometry_msgs::PointStamped point_laser;

    point_laser.header.frame_id = "laser";
    point_laser.header.stamp = ros::Time::now();
    point_laser.point.x = 1.0;
    point_laser.point.y = 2.0;
    point_laser.point.z = 7.3;
    //1.添加休眠，保证在处理信息时，base_link数据已经准备好了
    // ros::Duration(2).sleep();

    //5.转换算法实现
    ros::Rate rate(10);
    while (ros::ok())
        {   //调用了buffer的转换函数transform
            //参数1：被转换的座标点
            //参数2：目标坐标系
            //输出值：转换后的坐标点

            //PS1：包含头文件tf2_geometry_msgs/tf2_geometry_msgs.h
            //PS2：运行时抛出异常，base_link不存在，
            //    原因：订阅数据是一个耗时操作，在调用transform函数时，base_link数据还未准备好，
            //    解决方法：1.添加休眠，保证在处理信息时，base_link数据已经准备好了
            //            2.使用try-catch捕获异常，并等待一段时间后再次尝试转换

            //2.
            try
            {       
                geometry_msgs::PointStamped point_base = buffer.transform(point_laser,"base_link");//point_laser为待转换的坐标点，"base_link"为目标坐标系-
                ROS_INFO("转换后的数据:(%.2f,%.2f,%.2f),参考的坐标系是:%s",
                point_base.point.x,
                point_base.point.y,
                point_base.point.z,
                point_base.header.frame_id.c_str()
                );
            }               
            catch (const std::exception& e)
            {
                ROS_INFO("异常消息:%s",e.what());
            }
        //新建一个坐标点，用于接收转换结果  
        // geometry_msgs::PointStamped point_base; 
        // point_base = buffer.transform(point_laser,"base_link");//point_laser为待转换的坐标点，"base_link"为目标坐标系-
  
        // ROS_INFO("转换后的数据:(%.2f,%.2f,%.2f),参考的坐标系是:%s",
        // point_base.point.x,
        // point_base.point.y,
        // point_base.point.z,
        // point_base.header.frame_id.c_str()
        // );
        //最后输出
        rate.sleep();  
        ros::spinOnce();
    }
    return 0;
}
